# ModelCHECK Rules for Bulkhead Modules
# Rules for validating bulkhead CAD models

## Naming Rules
CHECK_NAMING_PATTERN: 53-10-BH-STA*
CHECK_FILE_PREFIX: 53-10-BH-STA
CHECK_FILE_SUFFIX: _MASTER_ASM | _SKEL | _ENV | _PANEL | _GUSSETS

## Structure Rules
REQUIRE_SKELETON: true
REQUIRE_ENVIRONMENT: true
NO_FEATURES_ON_ASM: true
SKELETON_OWNS_CSYS: true
SKELETON_OWNS_DATUMS: true

## Coordinate System Rules
REQUIRE_CSYS: A/C_ORIGIN, BH_STA*_CS
REQUIRE_DATUMS: FS, BL, WL
CHECK_DATUM_NAMING: FS_BL_WL.cs

## Interface Rules
IFC_DIRECTORY_REQUIRED: true
IFC_SUBDIRS: DATUMS, HOLE_PATTERNS, SEAL_SURFACES
EXPORT_FORMATS: STEP, JT, PDF

## Metadata Rules
REQUIRE_METADATA: part_number, revision, material, mass
CHECK_UNITS: mm
CHECK_BASELINE: Q100

## Quality Rules
MAX_FILE_SIZE_MB: 500
REQUIRE_CHANGELOG: true
REQUIRE_DRAWINGS: true
CHECK_EXPORT_VALIDITY: true
